N = 200
X = np.random.normal(loc=0, scale=1, size=(N,2))

# generate classes
# let the class be -1 under the mirror of the hyperbolic cosine function
# and +1 above
y = np.where(X[:,1] <= -1 * np.cosh(X[:,0]) + 1.5, -1, 1)
X[y == 1,1] += 0.5

clf = VotedPerceptron(eta=0.5, epoch=1)
clf.fit(X, y)

anim = clf.plot_animation(interval=400, notebook=True, smoothness=400)
HTML(anim.to_html5_video())
